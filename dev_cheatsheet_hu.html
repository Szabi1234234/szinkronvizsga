Táblák & adatok kezelése
Tábla létrehozása
CREATE TABLE felhasznalok (
  id       INT AUTO_INCREMENT PRIMARY KEY,
  nev      VARCHAR(100) NOT NULL,
  email    VARCHAR(255) UNIQUE NOT NULL,
  kor      INT,
  letrehozva DATETIME
    DEFAULT CURRENT_TIMESTAMP
);
Tábla módosítása

  ADD COLUMN varos VARCHAR(100);

  MODIFY COLUMN kor TINYINT UNSIGNED;

ALTER TABLE felhasznalok
  DROP COLUMN varos;

SELECT f.nev, r.cim
  FROM felhasznalok f
  INNER JOIN rendelesek r
    ON f.id = r.felhaszn_id;

-- LEFT JOIN: minden bal sor + egyező jobb
SELECT f.nev, r.cim
  FROM felhasznalok f
  LEFT JOIN rendelesek r
    ON f.id = r.felhaszn_id;
Aggregáció & GROUP BY
SELECT
  varos,
  COUNT(*) AS db,
  AVG(kor) AS atlag_kor,
  MAX(kor) AS max_kor,
  MIN(kor) AS min_kor,
  SUM(fizetes) AS ossz
FROM felhasznalok
GROUP BY varos
HAVING COUNT(*) > 5
ORDER BY db DESC;
